export default async function handler(req, res) { if (req.method !== 'POST') { return res.status(405).json({ error: 'Method not allowed' }); } const { input } = req.body; const apiKey = process.env.GEMINI_API_KEY; if (!apiKey) { return res.status(500).json({ error: 'Server API Key not configured' }); } let contentToAnalyze = input; // 檢查使用者輸入是否為網址 (URL) const urlRegex = /^https?:\/\//i; if (urlRegex.test(input.trim())) { try { // 嘗試抓取該網址的網頁內容 const webResponse = await fetch(input.trim(), { headers: { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36' } }); const htmlText = await webResponse.text(); // 簡單過濾 HTML 標籤，取出純文字作為 AI 的分析素材（實際應用中有些強力反爬網站可能需要專門的 API，但對多數公開公告與 YouTube 標題有效） contentToAnalyze = `使用者提供的目標網址: ${input}\n網頁提取內容摘要: ${htmlText.replace(/<[^>]*>?/gm, ' ').substring(0, 8000)}`; } catch (err) { // 如果該網站有嚴格的反爬蟲（例如 IG / X 登入牆），則退而求其次讓 AI 根據網址本身或提示進行分析 contentToAnalyze = `使用者提供的網址 (無法直接讀取詳細內容，請根據網址結構與常識推測或請使用者補充): ${input}`; } } const systemPrompt = ` 你是一個專業的 K-pop 追星情報分析小幫手。 請分析以下粉絲提供的內容（可能是網址或公告文字），提取重點，並**嚴格依照以下 JSON 格式輸出**，不要包含任何其他文字或 Markdown 標記 (\`\`\`json 也不要)。 格式要求： { "platform": "判斷情報來源 (例如: X/推特, Weverse, YouTube, 官方公告)，限制 6 個字以內", "points": ["用繁體中文列出3點最重要的摘要，每點不超過25字", "重點2", "重點3"], "reminders": [ {"label": "擷取事件名稱 (例如: 搶票, 預購, 開放登記)", "time": "YYYY/MM/DD HH:MM (若無具體時間請寫 未定)"} ], "expenseItem": "推測可能會花錢的品項名稱 (例如: 迷你七輯, 台北演唱會門票)", "expenseCategory": "必須從 [門票, 專輯, 周邊, 應援] 之中選擇一個最符合的類別", "expenseAmount": 推測金額的數字 (只要數字，例如 850。若沒寫金額請自行合理瞎猜一個大概數字，不要填 0) } 分析內容： ${contentToAnalyze}`; try { const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: systemPrompt }] }] }) }); const data = await response.json(); return res.status(200).json(data); } catch (error) { return res.status(500).json({ error: error.message }); } }
